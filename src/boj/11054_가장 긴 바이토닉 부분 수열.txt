# 40m
n = int(input())
nArr = list(map(int, input().split()))
rnArr = nArr[::-1]
d = [1]*n # 증가
d2 = [1]*n # 감소
for i in range(n):
    for j in range(i):
        if nArr[j] < nArr[i]:
            d[i] = max(d[i], d[j] + 1)
        if rnArr[j] < rnArr[i]:
            d2[i] = max(d2[i], d2[j] + 1)
ans = [0]*n
for i in range(n):
    ans[i] = d[i] + d2[n-1-i] -1

print(max(ans))
