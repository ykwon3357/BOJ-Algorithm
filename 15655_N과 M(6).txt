# 8m
n, m = map(int, input().split())
numList = sorted(list(map(int, input().split())))
visited = [0] * n
arr = [0] * m
def backtracking(k, idx):
    if k == m:
        print(*arr)
        return
    
    for i in range(idx, n):
        if not visited[i]:
            arr[k] = numList[i]
            visited[i] = 1
            backtracking(k+1, i)
            visited[i] = 0

backtracking(0, 0)
