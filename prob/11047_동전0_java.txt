import java.util.Scanner;

public class Main {
	
	public static void main(String[] args) {
		// 입력
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int k = sc.nextInt();
		int[] coin = new int[n];
		for(int i=0; i<n; i++) {
			coin[i] = sc.nextInt();
		}
		
		// 동전가치 큰거부터 나누면 개수는 최소
		int cnt = 0;
		for(int i=n-1; i>=0; i--) {
			if(k/coin[i] != 0) {
				cnt += k/coin[i];
				k %= coin[i];
			}
		}
		System.out.println(cnt);
	}
	
}
