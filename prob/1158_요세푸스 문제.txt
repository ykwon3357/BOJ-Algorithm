n, k = map(int, input().split())
data = [i for i in range(1, n+1)]
idx = k-1
l = len(data)
ans = []
while True:
    ans.append(data.pop(idx))
    if not data:
        break
    l -= 1
    idx = (idx+k-1)%l 

print('<',end='')
for i in range(len(ans)):
    if i == len(ans)-1:
        print(str(ans[i])+'>')
    else:
        print(str(ans[i])+',', end = ' ')
