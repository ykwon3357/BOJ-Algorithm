while True:
    try:
        string = input()

        data = [0]*4
        for s in string:
            if s.islower():
                data[0] += 1
            if s.isupper():
                data[1] += 1
            if s.isdigit():
                data[2] += 1
            if s.isspace():
                data[3] += 1
        print(*data)

    except EOFError:
        break
