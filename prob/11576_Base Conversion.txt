  6  2  
'''
a, b = map(int, input().split())
m = int(input())
nums = list(map(int, input().split()))

# a진법을 10진수로
dec = 0
for i in range(m):
    dec += nums[i]*a**(m-1-i)

# 10진수를 b진법으로
res = []
while dec:
    res.append(dec%b)
    dec//=b

print(*res[::-1])
