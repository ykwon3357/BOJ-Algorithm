n = int(input())
sg_card = list(map(int, input().split()))
m = int(input())
card = list(map(int, input().split()))

d = {}
for i in sg_card:
    if i in d:
        d[i] += 1
    else:
        d[i] = 1

for i in card:
    if i in d:
        print(d[i], end=' ')
    else:
        print(0, end=' ')
