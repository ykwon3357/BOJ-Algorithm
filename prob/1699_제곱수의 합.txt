'''
60m
  0 1 2 3 4 5 6 7
d 0 1 2 3 4 5 6 7 1의 제곱수 개수로 초기화

i 4, j 2일 때 d[4] = 4 > d[4 - 2 * 2] + 1 = d[0] + 1 = 1
d[4] = 1

i 5, j 2일 때 d[5] = 5 > d[5- 2*2] + 1 = d[1] + 1 = 2
dp[5] = 2

d[9] = d[9 - 3*3] +1 = d[0]+1 = 1
d[10] = d[10 - 3*3] +1 = d[1]+1 = 2
'''

n = int(input())
d = [i for i in range(n+1)]

for i in range(1, n+1):
    for j in range(1, i):
        if j*j > i:
            break
        if d[i] > d[i-j*j] + 1:
            d[i] = d[i-j*j] + 1
print(d[n])
