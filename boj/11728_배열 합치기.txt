n, m = map(int, input().split())
arr_a = list(map(int, input().split()))
arr_b = list(map(int, input().split()))
res = []

i, j = 0, 0
while i<n and j<m:
    
    if arr_a[i] < arr_b[j]:
        res.append(arr_a[i])
        i += 1
    else:
        res.append(arr_b[j])
        j += 1
        
if i < n:
    for i in range(i, n):
        res.append(arr_a[i])
else:
    for j in range(j, m):
        res.append(arr_b[j])

print(*res)
