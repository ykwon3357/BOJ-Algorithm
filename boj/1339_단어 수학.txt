'''
알파벳의 자릿수가 중요
딕셔너리 이용
들어오는 수 위치에 따라 10의 제곱수 곱합
'''
n = int(input())
words = [list(input()) for _ in range(n)]
d = {}

# (알파벳 : 알파벳의 자릿수 총합) 저장
for word in words:
    for i in range(len(word)):
        if word[i] in d:
            d[word[i]] += 10**(len(word)-1-i)
        else:
            d[word[i]] = 10**(len(word)-1-i)

# 알파벳 자릿수 높은수로 정렬 후 차례로 9부터 곱해서 결과값 구함
num, res = 9, 0
for i in sorted(d.values(), reverse=True):
    res += num*i
    num -= 1

print(res)
