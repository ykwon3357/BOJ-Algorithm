import sys

n = int(sys.stdin.readline().rstrip())
nArr = [int(sys.stdin.readline().rstrip()) for _ in range(n)]
stack = []
idx = 0
ans = []
for i in range(1, n+1):
    stack.append(i)
    ans.append('+')
    while stack and stack[len(stack)-1] == nArr[idx]:
        stack.pop()
        idx += 1
        ans.append('-')

if stack:
    print('NO')
else:
    print('\n'.join(ans))
