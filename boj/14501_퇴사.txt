# 뒤에서부터 dp
# day        3 4 5 6 7
# data[i][0] 1 1 2 4 2
# data[i][1] 10 20 15 40 200
# d          45 35 15 0 0 
n = int(input())
data = [list(map(int, input().split())) for _ in range(n)]

d = [0] * (n+1)

for i in range(n-1, -1, -1):
    if data[i][0] + i > n:
        d[i] = d[i+1]
    else:
        d[i] = max(d[i+1], data[i][1] + d[i+data[i][0]])
print(d[0])
