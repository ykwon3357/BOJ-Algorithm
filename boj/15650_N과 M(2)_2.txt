n, m = map(int, input().split())

s = []

def backtracking(idx):
    if len(s) == m:
        print(' '.join(map(str,s)))
        return
    
    for i in range(idx, n+1):
        if i in s:
            continue
        s.append(i)
        backtracking(i+1)
        s.pop()

backtracking(1)
