'''
20m
3 1 4
11 001 100
'''

def toBin(n):
    res = ''
    while n:
        res = str(n%2) + res
        n = n//2
    while len(res) < 3:
        res = '0' + res  
    return res

n = input()
ans = ''
for i in range(len(n)):  
    ans += toBin(int(n[i]))
    if not i:
        ans = str(int(ans))
print(ans)
